627 std::pair<iterator, bool> insert(const T& value) {
628 std::vector<Node*> update(MAX_LEVEL + 1);
629 Node* current = header;
630
631 // Find position and fill update array
632 for (int i = level; i >= 0; --i) {
633 while (current->forward[i] && comp(current->forward[i]->data, value)) {
634 current = current->forward[i];
635 }
636 update[i] = current;
637 }
638
639 current = current->forward[0]; // Move to data level
640
641 // Check if value already exists
642 if (current && !comp(current->data, value) && !comp(value, current->data)) {
643 return std::make_pair(iterator(current, header), false); // Element already exists
644 }
645
646 // Insert new node
647 int newLevel = randomLevel();
648 if (newLevel > level) {
649 for (int i = level + 1; i <= newLevel; ++i) {
650 update[i] = header; // New levels point from header
651 }
652 level = newLevel; // Update max level of the list
653 }
654
655 Node* newNode = createNode(value, newLevel);
656 for (int i = 0; i <= newLevel; ++i) {
657 newNode->forward[i] = update[i]->forward[i];
658 update[i]->forward[i] = newNode;
659 }
660
661 // Update backward pointers
662 newNode->backward = update[0]; // Previous node at level 0
663 if (newNode->forward[0]) {
664 newNode->forward[0]->backward = newNode; // Next node at level 0 points back to newNode
665 } else {
666 header->backward = newNode; // If newNode is the new last element, header's backward points to it
667 }
668
669 ++size_;
670 return std::make_pair(iterator(newNode, header), true);
671 }

803 iterator erase(const_iterator pos) {
804 if (pos == end()) {
805 // throw std::out_of_range("Cannot erase end() iterator."); // Or handle as desired
806 return end();
807 }
808
809 const T& value = *pos; // Get value to search for deletion
810
811 std::vector<Node*> update(MAX_LEVEL + 1);
812 Node* current = header;
813
814 // Find the node to delete and update pointers
815 for (int i = level; i >= 0; --i) {
816 while (current->forward[i] && comp(current->forward[i]->data, value)) {
817 current = current->forward[i];
818 }
819 update[i] = current;
820 }
821
822 current = current->forward[0]; // Node at level 0
823
824 // Check if the found node matches the value from the iterator (handles potential duplicates or not found if iterator is invalid)
825 if (!current || comp(current->data, value) || comp(value, current->data)) {
826 // This case should ideally not happen if 'pos' is a valid iterator to an element
827 // that hasn't been modified or deleted by other means.
828 return end(); // Element not found or iterator was invalid for erase by value.
829 }
830
831 Node* next_node = current->forward[0]; // Node immediately following 'current'
832
833 // Remove node from all levels
834 for (int i = 0; i <= level; ++i) {
835 if (update[i]->forward[i] == current) { // Only update if current is actually in this level
836 update[i]->forward[i] = current->forward[i];
837 }
838 }
839
840 // Update backward pointers
841 Node* prev = current->backward;
842 if (next_node) {
843 next_node->backward = prev;
844 } else {
845 header->backward = prev; // If current was the last element, header's backward points to its previous.
846 }
847
848 destroyNode(current); // Deallocate node
849
850 // Adjust the list's max_level if necessary
851 while (level > 0 && !header->forward[level]) {
852 --level;
853 }
854
855 --size_;
856 return iterator(next_node, header);
857 }
